    **Part A: Create `frontend/src/components/stats/TrendChart.vue`**

    A uPlot-based line chart component displaying mastery score and accuracy trends:

    ```vue
    <template>
      <div ref="chartRef" class="trend-chart-container" />
    </template>

    <script setup lang="ts">
    import { computed, onMounted, onBeforeUnmount, ref, watch } from 'vue';
    import uPlot from 'uplot';
    import 'uplot/dist/uPlot.min.css';
    import type { TrendPoint } from '../../composables/useTrends';

    const props = defineProps<{ points: TrendPoint[] }>();

    const chartRef = ref<HTMLDivElement>();
    let uplotInstance: uPlot | null = null;

    const chartData = computed((): uPlot.AlignedData | null => {
      if (!props.points.length) return null;
      const timestamps = props.points.map(p => p.date_epoch);
      const mastery = props.points.map(p => Math.round(p.avg_mastery * 100));
      const accuracy = props.points.map(p => Math.round(p.accuracy * 100));
      return [timestamps, mastery, accuracy];
    });

    const chartOptions: uPlot.Options = {
      width: 600,
      height: 250,
      axes: [
        { label: '日期' },
        { label: '%', scale: '%', values: (_u, vals) => vals.map(v => v + '%') },
      ],
      series: [
        {},
        { label: '掌握度', stroke: '#66ffe5', width: 2 },
        { label: '正确率', stroke: '#4caf50', width: 2 },
      ],
    };

    function initChart(): void {
      if (!chartRef.value || !chartData.value) return;
      uplotInstance = new uPlot(chartOptions, chartData.value, chartRef.value);
    }

    function destroyChart(): void {
      uplotInstance?.destroy();
      uplotInstance = null;
    }

    watch(chartData, () => {
      if (uplotInstance && chartData.value) {
        uplotInstance.setData(chartData.value);
      } else if (!uplotInstance && chartData.value) {
        initChart();
      }
    });

    onMounted(() => {
      if (chartData.value) initChart();
    });

    onBeforeUnmount(() => {
      destroyChart();
    });
    </script>

    <style scoped>
    .trend-chart-container {
      width: 100%;
      overflow-x: auto;
    }
    </style>
    ```

    IMPORTANT:
    - uPlot renders on a `<canvas>` element. The `chartRef` div is the target container.
    - `chartData` converts TrendPoint[] to uPlot's `AlignedData` format: `[timestamps[], series1[], series2[]]`.
    - The `%` scale formatter appends '%' to axis values.
    - `onBeforeUnmount` MUST call `uplotInstance?.destroy()` to prevent memory leaks (Pitfall 3 from RESEARCH.md).
    - uPlot CSS is imported from the package: `import 'uplot/dist/uPlot.min.css'`.
    - The container uses `overflow-x: auto` for responsive behavior on narrow screens.
    - DO NOT use `uplot-vue` wrapper -- it adds an unnecessary dependency layer when direct uPlot API is simpler and more controllable.

    A horizontal bar chart showing per-domain quiz accuracy. Uses `StatsNode.domain_tag` (added to the interface in Task 1) for grouping. The backend `/quiz-stats` endpoint returns `domain_tag` on each node (fixed in Plan 02).

    ```vue
    <template>
      <div class="domain-heatmap">
        <div v-for="d in domains" :key="d.domain" class="domain-row">
          <span class="domain-name">{{ d.domain || '未分类' }}</span>
          <div class="domain-bar-track">

            />
          </div>
          <span class="domain-pct">{{ d.accuracy }}%</span>
          <span class="domain-count">{{ d.correct }}/{{ d.total }}</span>

        </div>

    <script setup lang="ts">
    import { computed } from 'vue';
    import type { StatsNode } from '../../composables/useStats';

    const props = defineProps<{ nodes: StatsNode[] }>();

    interface DomainStats {
      domain: string;
      correct: number;
      total: number;
      accuracy: number;
    }

    const domains = computed<DomainStats[]>(() => {
      // Group nodes by domain_tag from the StatsNode interface.
      // domain_tag is set by tag_service_sqlite.py on the backend nodes table
      // and returned by get_quiz_stats_sqlite (domain_tag added in Plan 02).
      // Falls back to '未分类' when domain_tag is undefined/null.
      const map = new Map<string, { masterySum: number; count: number }>();
      for (const n of props.nodes) {
        const domain = n.domain_tag || '未分类';
        const entry = map.get(domain) || { masterySum: 0, count: 0 };
        entry.masterySum += n.mastery_score;
        entry.count += 1;
        map.set(domain, entry);
      }
      return Array.from(map.entries())
        .map(([domain, stats]) => ({
          domain,
          correct: Math.round(stats.masterySum * stats.count),
          total: stats.count * 100,
          accuracy: Math.round(stats.masterySum / stats.count * 100),
        }))
        .sort((a, b) => b.accuracy - a.accuracy);
    });
    </script>
    ```

    **Part C: Create `frontend/src/components/stats/ReviewHeatmap.vue`**

    A GitHub-style calendar heatmap (52 weeks x 7 days):

    ```vue
    <template>
      <div class="heatmap">
        <div v-for="(week, wi) in weeks" :key="wi" class="heatmap-week">
          <div
            v-for="(day, di) in week"
            :key="di"
            class="heatmap-day"
            :style="{ backgroundColor: colorForCount(day.count) }"
            :title="day.date + ': ' + day.count + ' 次复习'"
            :aria-label="day.date + ': ' + day.count + ' 次复习'"
          />
        </div>
      </div>
    </template>

    <script setup lang="ts">
    import { computed } from 'vue';
    import type { HeatmapDay } from '../../composables/useHeatmap';

    const props = defineProps<{ days: HeatmapDay[] }>();

    const weeks = computed(() => {
      const now = new Date();
      const oneYearAgo = new Date(now);
      oneYearAgo.setFullYear(oneYearAgo.getFullYear() - 1);

      const dayMap = new Map(props.days.map(d => [d.date, d.count]));
      const result: { date: string; count: number }[][] = [];
      const start = new Date(oneYearAgo);
      start.setDate(start.getDate() - start.getDay()); // align to Sunday

      for (let w = 0; w < 53; w++) {
        const week: { date: string; count: number }[] = [];
        for (let d = 0; d < 7; d++) {
          const date = new Date(start);
          date.setDate(date.getDate() + w * 7 + d);
          const dateStr = date.toISOString().slice(0, 10);
          week.push({ date: dateStr, count: dayMap.get(dateStr) || 0 });
        }
        result.push(week);
      }
      return result;
    });

    function colorForCount(count: number): string {
      if (count === 0) return 'rgba(255,255,255,0.06)';
      if (count <= 2) return 'rgba(102,255,229,0.22)';
      if (count <= 5) return 'rgba(102,255,229,0.44)';
      if (count <= 10) return 'rgba(102,255,229,0.66)';
      return 'rgba(102,255,229,0.88)';
    }
    </script>

    <style scoped>
    .heatmap {
      display: flex;
      gap: 2px;
      overflow-x: auto;
      padding: 8px 0;
    }

    .heatmap-week {
      display: flex;
      flex-direction: column;
      gap: 2px;
    }

    .heatmap-day {
      width: 12px;
      height: 12px;
      border-radius: 2px;
      transition: background-color 0.2s;
    }
    </style>
    ```

    IMPORTANT:
    - The grid layout follows the GitHub contribution graph convention: 53 columns (weeks) x 7 rows (days, Sunday=top).
    - The `start` date is aligned to the Sunday of the week containing the date one year ago.
