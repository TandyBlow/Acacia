    **Part A: Create `frontend/src/composables/useKeyboard.ts`**

    ```typescript
    import { onMounted, onBeforeUnmount } from 'vue';
    import { useNodeStore } from '../stores/nodeStore';

    export function useKeyboard() {
      const nodeStore = useNodeStore();

      function handleKeydown(e: KeyboardEvent): void {
        // Guard: don't intercept when typing in the TipTap editor
        if (e.target instanceof HTMLTextAreaElement || e.target instanceof HTMLInputElement) return;
        if ((e.target as HTMLElement)?.closest('.ProseMirror')) return;
        // Guard: don't intercept when a modal or dialog is open
        if ((e.target as HTMLElement)?.closest('[role="dialog"]')) return;
        // Guard: don't intercept when a Naive UI popover/modal is active
        if (document.querySelector('.n-modal-mask')) return;

        if (e.key === 'Escape') {
          e.preventDefault();
          if (nodeStore.isConfirmState || nodeStore.viewState !== 'display') {
            nodeStore.cancelOperation();
          }
          return;
        }

        if (e.ctrlKey && e.key === 'n') {
          e.preventDefault();
          if (nodeStore.viewState === 'display') {
            nodeStore.startAdd();
          }
        }
      }

      onMounted(() => window.addEventListener('keydown', handleKeydown));
      onBeforeUnmount(() => window.removeEventListener('keydown', handleKeydown));
    }
    ```

    IMPORTANT:
    - **ProseMirror guard** (`e.target.closest('.ProseMirror')`): This is CRITICAL. Without it, pressing Escape while typing in the editor would cancel the current operation instead of closing a TipTap popup (Pitfall 4 from RESEARCH.md). ProseMirror/TipTap renders inside a `<div class="ProseMirror" contenteditable="true">` which is NOT caught by `instanceof HTMLInputElement/HTMLTextAreaElement` checks.
    - **Dialog guard** (`[role="dialog"]`): Prevents keyboard shortcuts from firing when Radix Vue dialogs are open.
    - **Naive UI modal guard** (`.n-modal-mask`): Naive UI modals use this class as an overlay -- if present, don't handle shortcuts.
    - `Escape` calls `nodeStore.cancelOperation()` only when the store is in an active operation (`isConfirmState` or `viewState !== 'display'`). When in `display` mode, Escape is a no-op.
    - `Ctrl+N` calls `nodeStore.startAdd()` only when in `display` mode (idle). In any other mode, it's ignored.
    - The composable uses `onMounted`/`onBeforeUnmount` for automatic cleanup -- the caller just invokes `useKeyboard()` once.
    - DO NOT add keyboard shortcuts for Ctrl+E (toggle editor), Ctrl+Q (quiz), or Ctrl+R (review) -- those were listed as potential future shortcuts but are NOT in the phase requirements (UP-13 and UP-14 only cover Escape and Ctrl+N).
