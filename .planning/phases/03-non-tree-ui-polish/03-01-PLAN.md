    ```typescript
    import mermaid from 'mermaid';

    mermaid.initialize({
      startOnLoad: false,
      securityLevel: 'sandbox',
      theme: 'default',
    });

    let counter = 0;

    export async function renderMermaid(diagram: string): Promise<string> {
      const id = `mermaid-${++counter}-${Math.random().toString(36).slice(2, 10)}`;
      try {
        const { svg } = await mermaid.render(id, diagram);
        return svg;
      } catch (err) {
        console.error('Mermaid render error:', err);
        return '<div class="mermaid-error">图表渲染失败</div>';
      }
    }

    export function isMermaidLanguage(lang: string | null | undefined): boolean {
      if (!lang) return false;
      const normalized = lang.trim().toLowerCase();
      return normalized === 'mermaid' || normalized === 'mmd';
    }
    ```

    ```typescript
    function exportMarkdown(): void {
      if (!editor.value) return;
      const md = editor.value.getMarkdown();
      const blob = new Blob([md], { type: 'text/markdown' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      const safeName = (activeNode.value?.name ?? 'document').replace(/[/\\:*?"<>|]/g, '_');
      a.href = url;
      a.download = `${safeName}.md`;
      a.click();
      URL.revokeObjectURL(url);
    }

    function exportHTML(): void {
      if (!editor.value) return;
      const raw = editor.value.getHTML();
      const clean = DOMPurify.sanitize(raw, { USE_PROFILES: { html: true } });
      const safeName = (activeNode.value?.name ?? 'document').replace(/[/\\:*?"<>|]/g, '_');
      const fullDoc = `<!DOCTYPE html>
    <html lang="zh-CN">
    <head><meta charset="UTF-8"><title>${safeName}</title></head>
    <body>${clean}</body>
    </html>`;
      const blob = new Blob([fullDoc], { type: 'text/html' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `${safeName}.html`;
      a.click();
      URL.revokeObjectURL(url);
    }

    function exportPDF(): void {
      if (!editor.value) return;
      const rawHTML = editor.value.getHTML();
      const cleanHTML = DOMPurify.sanitize(rawHTML, { USE_PROFILES: { html: true } });
      const element = document.createElement('div');
      element.innerHTML = cleanHTML;
      const safeName = (activeNode.value?.name ?? 'document').replace(/[/\\:*?"<>|]/g, '_');
      const opt = {
        margin: [10, 10, 10, 10],
        filename: `${safeName}.pdf`,
        image: { type: 'jpeg', quality: 0.98 },
        html2canvas: { scale: 2, useCORS: true },
        jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' },
      };
      html2pdf().set(opt).from(element).save();
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Editor content -> Export file download | User-typed content (including potentially malicious HTML) crosses from TipTap editor to downloaded .html/.pdf file |
| Node name -> Export filename | The activeNode.name value (user-controlled) is used to construct the download filename |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-N3-01 | Information Disclosure | exportHTML(), exportPDF() in MarkdownEditor.vue | mitigate | Run `editor.getHTML()` output through `DOMPurify.sanitize(raw, { USE_PROFILES: { html: true } })` before creating the download blob. This prevents XSS in exported HTML/PDF files. |
| T-N3-02 | Tampering | exportMarkdown(), exportHTML(), exportPDF() in MarkdownEditor.vue | mitigate | Sanitize filenames derived from `activeNode.value?.name` using `.replace(/[/\\:*?"<>|]/g, '_')` to strip path traversal characters. This prevents directory traversal via malicious node names. |
| T-N3-03 | Information Disclosure | mermaidNode.ts `renderMermaid()` function | mitigate | Configure mermaid with `securityLevel: 'sandbox'` in `mermaid.initialize()`. This prevents script injection in SVG output from user-provided mermaid diagram source. |
</threat_model>
