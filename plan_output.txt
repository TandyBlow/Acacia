# Detailed Implementation Plan: 6-Step Tree Rendering System Refactoring

## Preamble: Architectural Context

The current `SceneManager.ts` (635 lines) is a monolithic class that handles camera, renderer, lights, tree mesh creation, ground, sky, theme application, animation, and resize logic. The refactoring touches nearly every subsystem. Each step must leave the tree visually functional so that incremental review is meaningful.

The Proctree vendor code (`frontend/src/vendor/proctree.ts`) has a `Branch` class with `head` (endpoint position as `Vec3`), `radius`, `child0`/`child1`, and `length` properties. private computeOrthoFrustum(w: number, h: number) {
```typescript
  if (!this.treeBounds) {
    // Default frustum before tree exists
    const halfH = 4;
    const halfW = halfH * (w / h);
    return { left: -halfW, right: halfW, top: halfH, bottom: -halfH };
  }
  const size = new THREE.Vector3();
  this.treeBounds.getSize(size);
  const padding = 1.3; // 30% padding
  const halfH = (size.y / 2) * padding;
  const halfW = (size.x / 2) * padding;
  const aspect = w / h;
  // Fit both axes, maintaining aspect ratio
  const frustumHalfH = Math.max(halfH, halfW / aspect);
  const frustumHalfW = frustumHalfH * aspect;
  return { left: -frustumHalfW, right: frustumHalfW, top: frustumHalfH, bottom: -frustumHalfH };
```

```typescript
export const ground2dVertexShader = /* glsl */ `
varying vec2 vScreenUV;

void main() {
  vScreenUV = uv;
  gl_Position = vec4(position.xy, 0.0, 1.0);
}
`;

export const ground2dFragmentShader = /* glsl */ `
uniform vec3 uGroundColor;
uniform float uGroundLineY;    // normalized 0-1, position of ground line on screen
uniform float uUndulation;     // amplitude of terrain contour
uniform float uTime;

varying vec2 vScreenUV;

// Simple 1D noise for contour
float hash(float n) { return fract(sin(n) * 43758.5453123); }
float noise1d(float x) {
  float i = floor(x);
  float f = fract(x);
  f = f * f * (3.0 - 2.0 * f);
  return mix(hash(i), hash(i + 1.0), f) * 2.0 - 1.0;
}

void main() {
  float y = vScreenUV.y;
  float groundLine = uGroundLineY;

  // Undulating contour: several octaves of noise offsetting the ground line
  float contour = groundLine
    + noise1d(vScreenUV.x * 8.0 + uTime * 0.05) * uUndulation * 0.03
    + noise1d(vScreenUV.x * 20.0 + 1.7) * uUndulation * 0.01;

  if (y > contour + 0.005) {
    // Above ground line: transparent (sky shows through)
    discard;
  } else if (y > contour - 0.003) {
    // The line itself: slightly darker contour stroke
    gl_FragColor = vec4(uGroundColor * 0.6, 1.0);
  } else {
    // Below ground line: solid dark brown fill
    gl_FragColor = vec4(uGroundColor, 1.0);
  }
}
`;
```

```typescript
export interface CrownSphere {
  geometry: THREE.IcosahedronGeometry;
  position: THREE.Vector3;
  radius: number;
}

export function buildCrownSpheres(
  terminals: TerminalBranch[],
  radiusCoefficient = 10
): CrownSphere[] {
  const spheres: CrownSphere[] = [];
  for (const t of terminals) {
    const sphereRadius = Math.max(0.05, t.radius * radiusCoefficient);
    const geo = new THREE.IcosahedronGeometry(sphereRadius, 2);
    spheres.push({
      geometry: geo,
      position: new THREE.Vector3(t.head[0], t.head[1], t.head[2]),
      radius: sphereRadius,
    });
  }
  return spheres;
}
```

#### A. New shader file `frontend/src/components/tree/shaders/crown.ts`

```typescript
import { SIMPLEX_3D } from './glslNoise';

export const crownVertexShader = SIMPLEX_3D + /* glsl */ `
uniform float uSphereRadius;
uniform float uDisplacementAmplitude;  // fraction of radius, ~0.15
uniform float uNoiseFrequency;         // low freq for displacement

varying vec3 vWorldPos;
varying vec3 vNormal;
varying vec3 vOrigNormal;

void main() {
  vOrigNormal = normalize(normalMatrix * normal);

  // Displacement along normal using 3D Perlin noise
  float noise = snoise(position * uNoiseFrequency);
  float displacement = noise * uSphereRadius * uDisplacementAmplitude;
  vec3 displaced = position + normal * displacement;

  // Approximate normal recalculation via finite differences
  float eps = 0.01;
  vec3 tangent1 = normalize(cross(normal, vec3(0.0, 1.0, 0.0)));
  if (length(tangent1) < 0.001) tangent1 = normalize(cross(normal, vec3(1.0, 0.0, 0.0)));
  vec3 tangent2 = normalize(cross(normal, tangent1));

  vec3 neighbor1 = position + tangent1 * eps;
  vec3 neighbor2 = position + tangent2 * eps;
  float d1 = snoise(neighbor1 * uNoiseFrequency) * uSphereRadius * uDisplacementAmplitude;
  float d2 = snoise(neighbor2 * uNoiseFrequency) * uSphereRadius * uDisplacementAmplitude;
  vec3 displaced1 = neighbor1 + normal * d1;
  vec3 displaced2 = neighbor2 + normal * d2;

  vec3 newNormal = normalize(cross(displaced1 - displaced, displaced2 - displaced));
  // Ensure normal faces outward
  if (dot(newNormal, normal) < 0.0) newNormal = -newNormal;

  vNormal = normalize(normalMatrix * newNormal);

  vec4 worldPos = modelMatrix * vec4(displaced, 1.0);
  vWorldPos = worldPos.xyz;

  gl_Position = projectionMatrix * modelViewMatrix * vec4(displaced, 1.0);
}
`;

export const crownFragmentShader = SIMPLEX_3D + /* glsl */ `
uniform vec3 uDarkGreen;
uniform vec3 uMidGreen;
uniform vec3 uLightGreen;
uniform float uLowFreqNoiseScale;
uniform float uHighFreqNoiseScale;
uniform float uHighFreqColorMix;

uniform float uEdgeFresnelThreshold;
uniform float uEdgeNoiseFrequency;
uniform float uEdgeDiscardThreshold;

uniform vec3 uLightDir;
uniform float uWarmShift;
uniform float uCoolShift;
uniform float uWarmBandWidth;

uniform vec3 uCameraPosition;

varying vec3 vWorldPos;
varying vec3 vNormal;
varying vec3 vOrigNormal;

void main() {
  vec3 normal = normalize(vNormal);

  // --- 1. Color blending ---
  float lowFreqNoise = snoise(vWorldPos * uLowFreqNoiseScale) * 0.5 + 0.5;
  float highFreqNoise = snoise(vWorldPos * uHighFreqNoiseScale) * 0.5 + 0.5;

  // Low freq controls dark/light zones
  vec3 baseColor = mix(uDarkGreen, uMidGreen, smoothstep(0.3, 0.5, lowFreqNoise));
  baseColor = mix(baseColor, uLightGreen, smoothstep(0.55, 0.75, lowFreqNoise));

  // High freq adds fine variation
  vec3 fineColor = mix(baseColor * 0.85, baseColor * 1.15, highFreqNoise * uHighFreqColorMix);

  // --- 2. Edge leaf fragmentation ---
  vec3 viewDir = normalize(uCameraPosition - vWorldPos);
  float fresnel = dot(viewDir, normal);

  if (fresnel < uEdgeFresnelThreshold) {
    float edgeNoise = snoise(vWorldPos * uEdgeNoiseFrequency) * 0.5 + 0.5;
    if (edgeNoise < uEdgeDiscardThreshold) {
      discard;
    }
  }

  // --- 3. Toon lighting ---
  float NdotL = dot(normalize(uLightDir), normal);
  // 2-step hard cutoffs
  float toonFactor;
  if (NdotL < 0.3) {
    toonFactor = 0.4;
  } else if (NdotL < 0.7) {
    toonFactor = 0.75;
  } else {
    toonFactor = 1.0;
  }

  // Warm/cool shift
  vec3 warmTint = vec3(uWarmShift, uWarmShift * 0.6, 0.0);
  vec3 coolTint = vec3(0.0, 0.0, uCoolShift);

  vec3 litColor;
  if (NdotL > 0.65 && NdotL < 0.75) {
    // Narrow warm transition band at light/shadow boundary
    litColor = fineColor * toonFactor + warmTint * uWarmBandWidth;
  } else if (NdotL >= 0.75) {
    litColor = fineColor * toonFactor + warmTint * 0.5;
  } else {
    litColor = fineColor * toonFactor + coolTint;
  }

  // --- 4. Final output ---
  gl_FragColor = vec4(litColor, 1.0);
}
`;
```

```typescript
import type { TreeParams } from '../../../vendor/proctree';

function hashUUID(uuid: string): number {
  let hash = 0;
  for (let i = 0; i < uuid.length; i++) {
    const char = uuid.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;  // Convert to 32-bit integer
  }
  return Math.abs(hash);
}

function lerp(a: number, b: number, t: number): number {
  return a + (b - a) * Math.max(0, Math.min(1, t));
}

function clamp(val: number, min: number, max: number): number {
  return Math.max(min, Math.min(max, val));
}

export function mapUserDataToTreeParams(
  nodeCount: number,
  maxDepth: number,
  widthDepthRatio: number,
  maxSubtreeRatio: number,
  userUUID: string,
): TreeParams {
  // levels = clamp(floor(log2(nodeCount)), 3, 6)
  const levels = clamp(Math.floor(Math.log2(Math.max(1, nodeCount))), 3, 6);

  // trunkLength = lerp(1.5, 4, log(nodeCount)/log(500))
  const trunkT = Math.log(Math.max(1, nodeCount)) / Math.log(500);
  const trunkLength = lerp(1.5, 4, trunkT);

  // climbRate = lerp(0.2, 0.7, widthDepthRatio < 1 ? 0.8 : 0.3)
  const climbRate = lerp(0.2, 0.7, widthDepthRatio < 1 ? 0.8 : 0.3);

  // sweepAmount = lerp(-0.3, 0.3, maxSubtreeRatio - 0.5)
  const sweepAmount = lerp(-0.3, 0.3, maxSubtreeRatio - 0.5);

  // seed from UUID hash
  const seed = hashUUID(userUUID) % 10000;

  return {
    levels,
    trunkLength,
    climbRate,
    sweepAmount,
    seed,

    // Fixed aesthetic safe values
    branchFactor: 3,
    initialBranchLength: 0.5,
    lengthFalloffFactor: 0.85,
    lengthFalloffPower: 0.5,
    maxRadius: 0.15,
    radiusFalloffRate: 0.7,
    taperRate: 0.9,
    trunkKink: 0.05,
    treeSteps: 6,
    twistRate: 3,
    segments: 8,
    trunkForks: 0,
    clumpMax: 0.5,
    clumpMin: 0.3,
    twigScale: 0.4,  // Irrelevant after Step 2, but kept for API compat
  };
}
```

#### B. Compute aggregate stats from existing data

```typescript
// Add to UserDataMapper class:
private lastNodeCount = 0;
private lastMaxDepth = 0;
private lastWidthDepthRatio = 1;
private lastMaxSubtreeRatio = 0.5;

update(statsNodes: StatsNode[], branches: Branch[], distribution: Record<string, number>) {
  // ... existing code ...

  // Compute aggregate stats
  this.lastNodeCount = statsNodes.length;
  this.lastMaxDepth = statsNodes.length > 0
    ? Math.max(...statsNodes.map(n => n.depth))
    : 0;

  // widthDepthRatio: ratio of max width (nodes at any single depth) to total depth
  const depthCounts = new Map<number, number>();
  for (const n of statsNodes) {
    depthCounts.set(n.depth, (depthCounts.get(n.depth) ?? 0) + 1);
  }
  const maxWidth = depthCounts.size > 0
    ? Math.max(...depthCounts.values())
    : 1;
  this.lastWidthDepthRatio = this.lastMaxDepth > 0
    ? maxWidth / this.lastMaxDepth
    : 1;

  // maxSubtreeRatio: largest subtree / total nodes
  const maxDesc = branches.length > 0
    ? Math.max(...branches.map(b => b.descendants ?? 0))
    : 0;
  this.lastMaxSubtreeRatio = statsNodes.length > 0
    ? maxDesc / statsNodes.length
    : 0.5;
}

getNodeCount() { return this.lastNodeCount; }
getMaxDepth() { return this.lastMaxDepth; }
getWidthDepthRatio() { return this.lastWidthDepthRatio; }
getMaxSubtreeRatio() { return this.lastMaxSubtreeRatio; }
```
